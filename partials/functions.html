Функции в JavaScript

Функции - это блоки кода, которые можно определить один раз и вызывать многократно для выполнения определенной задачи.

Объявление функций

Существует несколько способов объявления функций:

// Function Declaration
function greet(name) {
  console.log(`Привет, ${name}!`);
}

// Function Expression
const sayBye = function(name) {
  console.log(`Пока, ${name}!`);
};

// Arrow Function (стрелочная функция)
const add = (a, b) => {
  return a + b;
};

// Сокращенная стрелочная функция
const multiply = (a, b) => a * b;

greet("Иван");
sayBye("Мария");
console.log(add(2, 3)); // 5
console.log(multiply(4, 5)); // 20

Параметры и аргументы

Функции могут принимать параметры (переменные, перечисленные в определении функции) и аргументы (значения, передаваемые при вызове функции).

// Параметры по умолчанию
function createUser(name = "Гость", age = 18) {
  console.log(`Пользователь: ${name}, Возраст: ${age}`);
}

createUser("Алиса", 25); // Пользователь: Алиса, Возраст: 25
createUser("Боб");       // Пользователь: Боб, Возраст: 18
createUser();          // Пользователь: Гость, Возраст: 18

// Rest-параметр (...) для сбора оставшихся аргументов в массив
function sum(...numbers) {
  return numbers.reduce((total, num) => total + num, 0);
}

console.log(sum(1, 2, 3));    // 6
console.log(sum(10, 20, 30, 40)); // 100

Область видимости и замыкания

Область видимости определяет, где переменные доступны. Замыкания позволяют функции "запоминать" переменные из внешней области видимости.

function createCounter() {
  let count = 0; // Локальная переменная, захваченная замыканием
  return function() {
    count++;
    console.log(count);
  };
}

const counter = createCounter();
counter(); // 1
counter(); // 2
counter(); // 3
